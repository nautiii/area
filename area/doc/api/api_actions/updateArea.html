Future<void> updateArea(
    List<String> data, List<String> parameters, String id) async {
  Dio dio = new Dio();
  dio.options.headers["access-token"] = api.token;

  await dio.patch(
    'http://${conf.ip}:${conf.port}/api/user/action',
    data: {
      '_id': id,
      'action': {
        'name': data[1],
        'service': data[0],
        'param': data[2],
        'type': 'string',
        'input': parameters[0],
      },
      'reaction': {
        'name': data[4],
        'service': data[3],
        'param': data[5],
        'type': 'string',
        'input': parameters[1],
      },
    },
  );
}
